            //classe que formata números com duas casas decimais
          DecimalFormat df= new DecimalFormat("0.00");

                System.out.println("Para uma parcelas fica :"+preçoTotal );
            }
                else{
                if ((parcelas>=2)&&(parcelas<=5)) {                                                       //df.format() faz a formatação nas parcelas
                    preçoTotal=preçoTabela;
                   System.out.println("De 2 a 5 parcelas fica : "+preçoTotal + " são "+ parcelas +" de: "+df.format(calculaParcelas(preçoTotal) ));

                        System.out.println("De 6 a 10 parcelas fica : "+preçoTotal + " são "+ parcelas +" de: "+df.format(calculaParcelas(preçoTotal )));

        public double calculaParcelas(double preçoTotal){
